### Basic Usage

```python
from ssi_framework import SSIAgent, CausalReasoner, NeurosymbolicEngine

# Create a sovereign AI agent
agent = SSIAgent(
    name="sovereign_agent_1",
    causal_model="scm_config.yaml",
    neurosymbolic_engine="scallop",
    sovereignty_level=8.5
)

# Initialize causal reasoning
reasoner = CausalReasoner(
    structural_equations="./models/causal_graph.yaml"
)

# Perform interventional reasoning
result = reasoner.intervene(
    variable="treatment",
    value=1.0,
    outcome="effect"
)

print(f"Causal effect: {result.ate}")  # Average Treatment Effect

# Neurosymbolic inference
ns_engine = NeurosymbolicEngine(framework="scallop")
ns_engine.load_rules("./rules/logic_program.scl")
ns_engine.load_neural_model("./models/embedding_net.pt")

# Query with neural + symbolic fusion
answer = ns_engine.query(
    "path(A, B) :- edge(A, C), edge(C, B)",
    bindings={"A": "node_1"}
)

print(f"Inference result: {answer}")
```
